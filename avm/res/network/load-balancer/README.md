### Example 2: _Using internal load balancer parameter_

This instance deploys the module with the minimum set of required parameters to deploy an internal load balancer.


<details>

```bicep
module loadBalancer 'br/public:avm/res/network/load-balancer:<version>' = {
  name: 'loadBalancerDeployment'
  params: {
    // Required parameters
    frontendIPConfigurations: [
      {
        name: 'privateIPConfig1'
        subnetId: '<subnetId>'
      }
    ]
    name: 'nlbint001'
    // Non-required parameters
    backendAddressPools: [
      {
        name: 'servers'
      }
    ]
    inboundNatRules: [
      {
        backendPort: 443
        enableFloatingIP: false
        enableTcpReset: false

        frontendPort: 443

        protocol: 'Tcp'
      }

        frontendPort: 3389
        name: 'inboundNatRule2'

    ]
    loadBalancingRules: [
      {
        backendAddressPoolName: 'servers'
        backendPort: 0
        disableOutboundSnat: true
        enableFloatingIP: true
        enableTcpReset: false

        frontendPort: 0
        idleTimeoutInMinutes: 4
        loadDistribution: 'Default'
        name: 'privateIPLBRule1'
        probeName: 'probe1'
        protocol: 'All'
      }
    ]
    location: '<location>'
    probes: [
      {
        intervalInSeconds: 5
        name: 'probe1'
        numberOfProbes: 2
        port: '62000'
        protocol: 'Tcp'
      }
    ]
    roleAssignments: [
      {
        principalId: '<principalId>'
        principalType: 'ServicePrincipal'

      }
      {
```

| [`frontendIPConfigurations`](#parameter-frontendipconfigurations) | array | Array of objects containing all frontend IP configurations. |
